\graphicspath{ {./images/} }

\newcommand{\image}[4]{
  \begin{figure}[here]
    \centering
    \includegraphics[width=10cm]{#1}
    \caption[#3]{\emph{#4}}
    \label{fig:#2}
  \end{figure}
}

\newcommand{\subimage}[5]{
  \begin{subfigure}[t]{#5\textwidth}
    \includegraphics[width=\linewidth]{#1}
    \caption[#3]{\emph{#4}}
    \label{fig:#2}
  \end{subfigure}
}
